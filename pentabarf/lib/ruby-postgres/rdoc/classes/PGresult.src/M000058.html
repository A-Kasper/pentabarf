/*
 * call-seq:
 *    res.each{ |tuple| ... }
 *
 * Invokes the block for each tuple (row) in the result.
 *
 * Equivalent to <tt>res.result.each{ |tuple| ... }</tt>.
 */
static VALUE
pgresult_each(obj)
    VALUE obj;
{
    rb_iterate(rb_ary_each, pgresult_result(obj), rb_yield, 0);
    return obj;
}
